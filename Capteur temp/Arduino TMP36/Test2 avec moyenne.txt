int tempPin=A0;

float readTemp(int pin)

{
float bv=analogRead(pin);
float mv=bv*3.3/4096.0;
float temperatureC = (mv - 0.5) * 100 ;
return temperatureC;
}

float readAvgTemp(int pin, int waitDelay, int count)

{
int k=0;
float sum=0.0;
for(k=0;k<count;k++)
	{
sum=sum + readTemp(pin);
delay(waitDelay);
	}
float avg=sum/(float)count;
return avg;
}

void setup()

{
Serial.begin(9600);
}

void loop()

{
Serial.print("Temperature : ");
Serial.println(readAvgTemp(tempPin, 1, 100)); //Par exemple, readAvgTemp(A0, 1,100) retournera la moyenne de 100 mesures faites à 1ms d’écart.
delay(100);
}
